方法执行时间拦截器

实现代码

class MethodInterceptor {
    private timings: Map = new Map();
    
    // 拦截对象的所有方法
    intercept(obj: any, className: string = ''): any {
        const handler = {
            get: (target: any, prop: string, receiver: any) => {
                const original = target[prop];
                
                if (typeof original === 'function') {
                    return (...args: any[]) => {
                        const start = performance.now();
                        const result = original.apply(target, args);
                        
                        // 处理异步方法
                        if (result && typeof result.then === 'function') {
                            return result.then((res: any) => {
                                this.recordTiming(className, prop, start);
                                return res;
                            }).catch((error: any) => {
                                this.recordTiming(className, prop, start);
                                throw error;
                            });
                        }
                        
                        // 同步方法
                        this.recordTiming(className, prop, start);
                        return result;
                    };
                }
                
                return original;
            }
        };
        
        return new Proxy(obj, handler);
    }
    
    // 记录执行时间
    private recordTiming(className: string, methodName: string, startTime: number): void {
        const endTime = performance.now();
        const duration = endTime - startTime;
        const fullMethodName = className ? `${className}.${methodName}` : methodName;
        
        if (!this.timings.has(fullMethodName)) {
            this.timings.set(fullMethodName, []);
        }
        
        this.timings.get(fullMethodName)!.push(duration);
    }
    
    // 获取统计结果
    getStatistics(): Array<{method: string, calls: number, totalTime: number, avgTime: number}> {
        const stats: Array<{method: string, calls: number, totalTime: number, avgTime: number}> = [];
        
        this.timings.forEach((times, method) => {
            const totalTime = times.reduce((sum, time) => sum + time, 0);
            stats.push({
                method,
                calls: times.length,
                totalTime,
                avgTime: totalTime / times.length
            });
        });
        
        return stats.sort((a, b) => b.totalTime - a.totalTime);
    }
    
    // 重置统计数据
    reset(): void {
        this.timings.clear();
    }
    
    // 装饰器形式（用于TypeScript类方法）
    static measure(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
        const originalMethod = descriptor.value;
        
        descriptor.value = function (...args: any[]) {
            const start = performance.now();
            const result = originalMethod.apply(this, args);
            
            if (result && typeof result.then === 'function') {
                return result.then((res: any) => {
                    const end = performance.now();
                    console.log(`${propertyKey} 执行时间: ${(end - start).toFixed(2)}ms`);
                    return res;
                });
            }
            
            const end = performance.now();
            console.log(`${propertyKey} 执行时间: ${(end - start).toFixed(2)}ms`);
            return result;
        };
        
        return descriptor;
    }
}

// 示例类
class ExampleService {
    private interceptor: MethodInterceptor;
    
    constructor() {
        this.interceptor = new MethodInterceptor();
        // 拦截自身所有方法
        return this.interceptor.intercept(this, 'ExampleService');
    }
    
    heavyCalculation(iterations: number): number {
        let result = 0;
        for (let i = 0; i < iterations; i++) {
            result += Math.sqrt(i) * Math.sin(i);
        }
        return result;
    }
    
    async fetchData(delay: number): Promise {
        return new Promise(resolve => {
            setTimeout(() => {
                resolve(`Data fetched after ${delay}ms`);
            }, delay);
        });
    }
    
    getStats() {
        return this.interceptor.getStatistics();
    }
}
